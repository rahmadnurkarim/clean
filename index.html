Langkah 1: Ekstrak Data Pengikut & Diikuti (Dual-Scan)

Skrip ini berjalan langsung di browser Anda untuk mengunduh daftar **Pengikut** (*Followers*) dan **Diikuti** (*Following*) secara bersamaan, kemudian menggabungkannya ke dalam satu file terkompresi.

PENTING: Abaikan Pesan Error Merah/Kuning di Console!

Selama proses pemindaian berjalan di browser, Anda mungkin melihat peringatan merah seperti:
TRANSPORT_ESTABLISHMENT_TIMEOUT [Failed to establish lightspeed stream]
WebSocket connection failed: WebSocket is closed before the connection...

Kenapa ini terjadi? Error tersebut adalah gangguan internal server obrolan/DM real-time Instagram sendiri (Lightspeed Stream) di browser Laptop Anda.
Sama sekali tidak merusak skrip InstaClean! Skrip kita berjalan menggunakan jalur API HTTPS GraphQL yang berbeda dan sangat aman. Biarkan proses pemindaian tetap berjalan sampai selesai mengunduh berkas .json secara otomatis.
1. Buka Instagram Web

Masuk ke profil Instagram Anda menggunakan Google Chrome atau Edge di Laptop/PC.

2. Buka Console Browser

Tekan tombol F12 di keyboard Anda lalu klik tab **Console**.

3. Tempel & Jalankan

Salin skrip di bawah ini, tempelkan di area Console, lalu tekan **Enter**. Berkas JSON terunduh otomatis.

Instagram Dual GraphQL Scraper (Following & Followers) 
/* =========================================================================
 * INSTACLEAN WEB DUAL-SCANNER (FOLLOWERS & FOLLOWING)
 * Jalankan ini di console profil Instagram Anda
 * ========================================================================= */
(async () => {
    console.log("%c[InstaClean Scanner] Menghubungi GraphQL Instagram...", "color: #e0a33a; font-weight: bold; font-size: 14px;");
    
    const getCookie = (name) => {
        const value = `; ${document.cookie}`;
        const parts = value.split(`; ${name}=`);
        if (parts.length === 2) return parts.pop().split(';').shift();
        return null;
    };

    const targetUserId = getCookie("ds_user_id");
    if (!targetUserId) {
        console.error("%c[Error] Sesi login tidak ditemukan. Harap login kembali di Instagram!", "color: #ef6a62; font-weight: bold;");
        return;
    }

    const sleep = (ms) => new Promise(r => setTimeout(r, ms));
    
    let followingList = [];
    let followersList = [];
    
    // 1. SCAN FOLLOWING (DIKUTI)
    console.log("%c[Scan 1/2] Memulai Pemindaian Daftar Diikuti (Following)...", "color: #62d6d0; font-weight: bold;");
    let hasNextPage = true;
    let endCursor = '';
    let page = 0;
    
    while (hasNextPage) {
        page++;
        const variables = { id: targetUserId, include_reel: false, fetch_mutual: false, first: 50, after: endCursor };
        const queryHash = '3dec7e2c57367ef3da3d987d89f9dbc8';
        const url = `https://www.instagram.com/graphql/query/?query_hash=${queryHash}&variables=${encodeURIComponent(JSON.stringify(variables))}`;

        try {
            const res = await fetch(url).then(r => r.json());
            if (!res.data || !res.data.user || !res.data.user.edge_follow) {
                throw new Error("Format respons Following tidak valid.");
            }
            const data = res.data.user.edge_follow;
            data.edges.forEach(edge => {
                followingList.push({
                    id: edge.node.id,
                    username: edge.node.username,
                    full_name: edge.node.full_name || '',
                    profile_pic_url: edge.node.profile_pic_url,
                    is_private: edge.node.is_private,
                    is_verified: edge.node.is_verified
                });
            });
            console.log(` -> Memuat ${data.edges.length} Following. Total: ${followingList.length}`);
            hasNextPage = data.page_info.has_next_page;
            endCursor = data.page_info.end_cursor;
            await sleep(1000 + Math.random() * 1000);
        } catch (err) {
            console.error("Gagal memuat Following:", err);
            break;
        }
    }

    // 2. SCAN FOLLOWERS (PENGIKUT)
    console.log("%c[Scan 2/2] Memulai Pemindaian Daftar Pengikut (Followers)...", "color: #62d6d0; font-weight: bold;");
    hasNextPage = true;
    endCursor = '';
    page = 0;

    while (hasNextPage) {
        page++;
        const variables = { id: targetUserId, include_reel: true, fetch_mutual: true, first: 50, after: endCursor };
        const queryHash = 'c76146de99bb02f6415203be841dd25a';
        const url = `https://www.instagram.com/graphql/query/?query_hash=${queryHash}&variables=${encodeURIComponent(JSON.stringify(variables))}`;

        try {
            const res = await fetch(url).then(r => r.json());
            if (!res.data || !res.data.user || !res.data.user.edge_followed_by) {
                throw new Error("Format respons Followers tidak valid.");
            }
            const data = res.data.user.edge_followed_by;
            data.edges.forEach(edge => {
                followersList.push({
                    id: edge.node.id,
                    username: edge.node.username,
                    full_name: edge.node.full_name || '',
                    profile_pic_url: edge.node.profile_pic_url,
                    is_private: edge.node.is_private,
                    is_verified: edge.node.is_verified
                });
            });
            console.log(` -> Memuat ${data.edges.length} Followers. Total: ${followersList.length}`);
            hasNextPage = data.page_info.has_next_page;
            endCursor = data.page_info.end_cursor;
            await sleep(1000 + Math.random() * 1000);
        } catch (err) {
            console.error("Gagal memuat Followers:", err);
            break;
        }
    }

    // 3. MERGE & CROSS-REFERENCE
    console.log("%c[Scan Selesai] Melakukan Sinkronisasi Hubungan Teman...", "color: #8ccf7e; font-weight: bold;");
    const followerIds = new Set(followersList.map(f => f.id));
    const followingIds = new Set(followingList.map(f => f.id));

    // Gabungkan list unik
    const allUsersMap = new Map();
    
    followingList.forEach(u => {
        allUsersMap.set(u.id, {
            ...u,
            is_following: true,
            is_follower: followerIds.has(u.id)
        });
    });

    followersList.forEach(u => {
        if (!allUsersMap.has(u.id)) {
            allUsersMap.set(u.id, {
                id: u.id,
                username: u.username,
                full_name: u.full_name || '',
                profile_pic_url: u.profile_pic_url || '',
                is_private: u.is_private || false,
                is_verified: u.is_verified || false,
                is_following: false,
                is_follower: true
            });
        } else {
            const existing = allUsersMap.get(u.id);
            existing.is_follower = true;
            if (!existing.profile_pic_url && u.profile_pic_url) existing.profile_pic_url = u.profile_pic_url;
            if (!existing.full_name && u.full_name) existing.full_name = u.full_name;
            allUsersMap.set(u.id, existing);
        }
    });

    const finalOutput = Array.from(allUsersMap.values());
    console.log(`%c[Sukses] Berhasil memetakan total ${finalOutput.length} relasi unik!`, "color: #8ccf7e; font-weight: bold;");

    const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(finalOutput, null, 2));
    const dlAnchor = document.createElement('a');
    dlAnchor.setAttribute("href", dataStr);
    dlAnchor.setAttribute("download", `instaclean_dual_${targetUserId}.json`);
    document.body.appendChild(dlAnchor);
    dlAnchor.click();
    dlAnchor.remove();
})();
