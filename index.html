Langkah 1: Ekstrak Data dari Instagram Web

Skrip GraphQL di bawah ini berjalan langsung di browser Anda menggunakan session token aman Anda. Tidak ada data sandi yang dikirimkan ke luar.

1. Buka Instagram

Buka Instagram Desktop di browser Anda, masuk ke profil Anda sendiri.

2. Buka Console

Tekan F12 (Inspect) -> klik Tab **Console**.

3. Tempel & Jalankan

Salin skrip di bawah ini, tempel di console, tekan **Enter**. File JSON otomatis terunduh.

Instagram GraphQL Scraper Script (Query 3dec7e2c) 
/* =========================================================================
 * INSTACLEAN WEB-CONNS GRAPHQL SCANNER
 * Jalankan ini di console profil Instagram Anda
 * ========================================================================= */
(async () => {
    console.log("%c[InstaClean Scanner] Menghubungi server GraphQL Instagram...", "color: #e0a33a; font-weight: bold; font-size: 14px;");
    
    const getCookie = (name) => {
        const value = `; ${document.cookie}`;
        const parts = value.split(`; ${name}=`);
        if (parts.length === 2) return parts.pop().split(';').shift();
        return null;
    };

    const targetUserId = getCookie("ds_user_id");
    if (!targetUserId) {
        console.error("%c[Error] Session ID tidak ditemukan. Harap login kembali di Instagram!", "color: #ef6a62; font-weight: bold;");
        return;
    }

    let followingList = [];
    let endCursor = '';
    let hasNextPage = true;
    let pageCount = 0;

    while (hasNextPage) {
        pageCount++;
        const variables = {
            id: targetUserId,
            include_reel: true,
            fetch_mutual: false,
            first: 50,
            after: endCursor
        };

        const queryHash = '3dec7e2c57367ef3da3d987d89f9dbc8';
        const url = `https://www.instagram.com/graphql/query/?query_hash=${queryHash}&variables=${encodeURIComponent(JSON.stringify(variables))}`;

        try {
            const response = await fetch(url).then(res => res.json());
            const data = response.data.user.edge_follow;
            const edges = data.edges;

            edges.forEach(edge => {
                followingList.push({
                    id: edge.node.id,
                    username: edge.node.username,
                    full_name: edge.node.full_name || '',
                    profile_pic_url: edge.node.profile_pic_url,
                    is_private: edge.node.is_private,
                    is_verified: edge.node.is_verified,
                    follows_viewer: edge.node.follows_viewer || false // Status apakah dia follback Anda
                });
            });

            console.log(`[Scan Page ${pageCount}] Mengambil ${edges.length} pertemanan. Total data: ${followingList.length}`);
            
            hasNextPage = data.page_info.has_next_page;
            endCursor = data.page_info.end_cursor;

            // Random delay anti-block
            await new Promise(r => setTimeout(r, Math.floor(Math.random() * 1000) + 1000));
        } catch (err) {
            console.error("[Scanner] Error fetching GraphQL:", err);
            break;
        }
    }

    console.log("%c[Sukses] Mengunduh berkas instaclean_export.json...", "color: #8ccf7e; font-weight: bold;");
    const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(followingList, null, 2));
    const dlAnchor = document.createElement('a');
    dlAnchor.setAttribute("href", dataStr);
    dlAnchor.setAttribute("download", `instaclean_gql_${targetUserId}.json`);
    document.body.appendChild(dlAnchor);
    dlAnchor.click();
    dlAnchor.remove();
})();
